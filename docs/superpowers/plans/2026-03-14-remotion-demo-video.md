# Hybrid Remotion + Playwright Demo Video — Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a two-stage pipeline that captures screenshots and video clips from the live Sports2 app (Playwright), then composes them into a polished ~78-second demo video (Remotion).

**Architecture:** Stage 1 (this repo) refactors the existing `tests/demo/record-demo.cjs` into a headless asset capture script that outputs screenshots, .webm clips, and a `manifest.json`. Stage 2 (separate repo `sports2-demo-video`) is a standalone Remotion project that reads the manifest and composes scenes with animated HUD overlays, Ken Burns effects, and fade-to-black transitions into MP4 + WebM output.

**Tech Stack:** Playwright (asset capture), Remotion v4 (video composition), React + TypeScript (Remotion components)

**Spec:** `docs/superpowers/specs/2026-03-14-remotion-demo-video-design.md`

```javascript
/**
 * Capture a video clip in a fresh browser context.
 * Returns the actual duration in seconds.
 *
 * Uses Playwright's page.video().saveAs() API to reliably save the clip
 * to the desired path (avoids fragile directory-scanning heuristics).
 *
 * @param {import('playwright').Browser} browser
 * @param {object} storageState - Auth cookies/session from primary context
 * @param {string} clipPath - Output file path for the .webm clip
 * @param {function} interactionFn - Async function receiving (page) to perform interactions
 * @returns {Promise<number>} Actual clip duration in seconds
 */
async function captureClip(browser, storageState, clipPath, interactionFn) {
  const clipContext = await browser.newContext({
    viewport: { width: 1440, height: 900 },
    storageState,
    recordVideo: {
      dir: path.dirname(clipPath),
      size: { width: 1440, height: 900 },
    },
  });

  const clipPage = await clipContext.newPage();

  // Lock theme
  await clipPage.addInitScript(() => {
    document.cookie = 'vite-ui-theme=light; path=/; max-age=86400';
  });

  const startTime = Date.now();

  try {
    await interactionFn(clipPage);
  } catch (error) {
    console.log(`    ⚠ Clip interaction error: ${error.message}`);
  }

  const durationSec = Math.round((Date.now() - startTime) / 1000);

  // Save the video to the desired path using Playwright's built-in API
  // Must be called before page.close() to get the video handle
  const video = clipPage.video();
  await clipPage.close();
  if (video) {
    await video.saveAs(clipPath);
  }
  await clipContext.close();

  return durationSec;
}
```

```typescript
import manifestJson from '../../public/assets/manifest.json';

export interface ManifestScene {
  id: string;
  type: 'screenshot' | 'clip';
  file: string;
  durationSec: number;
  hud: {
    title: string;
    subtitle: string;
  };
}

export interface Manifest {
  scenes: ManifestScene[];
  outro: {
    durationSec: number;
    lines: string[];
  };
  transitionDurationSec: number;
}

export const FPS = 30;

export function getManifest(): Manifest {
  return manifestJson as Manifest;
}

export function secToFrames(sec: number): number {
  return Math.round(sec * FPS);
}

/**
 * Calculate the total duration of the video in frames.
 * scenes + transitions between each scene pair + transition before outro + outro
 */
export function getTotalFrames(manifest: Manifest): number {
  const sceneFrames = manifest.scenes.reduce(
    (sum, s) => sum + secToFrames(s.durationSec),
    0
  );
  const transitionCount = manifest.scenes.length; // between each pair + before outro
  const transitionFrames = transitionCount * secToFrames(manifest.transitionDurationSec);
  const outroFrames = secToFrames(manifest.outro.durationSec);
  return sceneFrames + transitionFrames + outroFrames;
}
```

import { AbsoluteFill, interpolate, useCurrentFrame } from 'remotion';

interface SceneTransitionProps {
  durationInFrames: number;
}

export const SceneTransition: React.FC<SceneTransitionProps> = ({
  durationInFrames,
}) => {
  const frame = useCurrentFrame();
  const half = durationInFrames / 2;

  // Fade in to black, hold, fade out from black
  const opacity = interpolate(
    frame,
    [0, half * 0.6, half, half * 1.4, durationInFrames],
    [0, 1, 1, 1, 0],
    { extrapolateLeft: 'clamp', extrapolateRight: 'clamp' }
  );

  return (
    <AbsoluteFill
      style={{
        backgroundColor: 'black',
        opacity,
      }}
    />
  );
};
```

interface HudOverlayProps {
  title: string;
  subtitle: string;
  durationInFrames: number;
}

export const HudOverlay: React.FC<HudOverlayProps> = ({
  title,
  subtitle,
  durationInFrames,
}) => {
  const frame = useCurrentFrame();
  const { fps } = useVideoConfig();

  // Animate in with spring at scene start
  const slideIn = spring({
    frame,
    fps,
    config: { damping: 15, stiffness: 120 },
  });

  // Animate out 1 second before scene ends
  const fadeOutStart = durationInFrames - fps; // 1 second before end
  const fadeOut = interpolate(
    frame,
    [fadeOutStart, durationInFrames],
    [1, 0],
    { extrapolateLeft: 'clamp', extrapolateRight: 'clamp' }
  );

  const opacity = Math.min(slideIn, fadeOut);
  const translateY = interpolate(slideIn, [0, 1], [20, 0]);

  return (
    <AbsoluteFill>
      <div
        style={{
          position: 'absolute',
          bottom: 24,
          left: 24,
          background: 'rgba(0, 0, 0, 0.75)',
          backdropFilter: 'blur(8px)',
          WebkitBackdropFilter: 'blur(8px)',
          borderRadius: 12,
          padding: '16px 24px',
          maxWidth: 400,
          boxShadow: '0 4px 24px rgba(0,0,0,0.3)',
          fontFamily:
            '-apple-system, BlinkMacSystemFont, "Segoe UI", Roboto, sans-serif',
          opacity,
          transform: `translateY(${translateY}px)`,
          zIndex: 10,
        }}
      >
        <div
          style={{
            fontSize: 20,
            fontWeight: 600,
            color: 'white',
            lineHeight: 1.3,
          }}
        >
          {title}
        </div>
        <div
          style={{
            fontSize: 14,
            color: 'rgba(255,255,255,0.7)',
            marginTop: 4,
          }}
        >
          {subtitle}
        </div>
      </div>
    </AbsoluteFill>
  );
};
```

- [ ] **Step 1: Create DemoVideo — the main composition**

Write `../sports2-demo-video/src/DemoVideo.tsx`:

```tsx
import { AbsoluteFill, Sequence } from 'remotion';

import { Scene } from './components/Scene';
import { SceneTransition } from './components/SceneTransition';
import { TechStackOutro } from './components/TechStackOutro';
import { getManifest, secToFrames } from './lib/manifest';

export const DemoVideo: React.FC = () => {
  const manifest = getManifest();
  const transitionFrames = secToFrames(manifest.transitionDurationSec);

  // Build timeline: scene, transition, scene, transition, ..., transition, outro
  let currentFrame = 0;
  const sequences: React.ReactNode[] = [];

  manifest.scenes.forEach((scene, index) => {
    const sceneFrames = secToFrames(scene.durationSec);

    // Scene
    sequences.push(
      <Sequence key={`scene-${scene.id}`} from={currentFrame} durationInFrames={sceneFrames}>
        <Scene scene={scene} />
      </Sequence>
    );
    currentFrame += sceneFrames;

    // Transition after every scene (including last — transitions into outro)
    sequences.push(
      <Sequence key={`transition-${index}`} from={currentFrame} durationInFrames={transitionFrames}>
        <SceneTransition durationInFrames={transitionFrames} />
      </Sequence>
    );
    currentFrame += transitionFrames;
  });

  // Outro
  const outroFrames = secToFrames(manifest.outro.durationSec);
  sequences.push(
    <Sequence key="outro" from={currentFrame} durationInFrames={outroFrames}>
      <TechStackOutro lines={manifest.outro.lines} durationInFrames={outroFrames} />
    </Sequence>
  );

  return (
    <AbsoluteFill style={{ backgroundColor: 'black' }}>
      {sequences}
    </AbsoluteFill>
  );
};
```
